Also, for your reference, here is the code for the example BankAccountTest:


    import static org.junit.Assert.*;
    import org.junit.Test;

    public class BankAccountTest {

    	@Test
    	public void testDeposit() {
    		BankAccount ba = new BankAccount(100.00);
    		ba.deposit(200.00);
    		assertEquals(300.00, ba.money, 0.001);
    	}

    	@Test
    	public void testWithdraw() {
    		BankAccount ba = new BankAccount(9000.00);
    		assertEquals(9000.00, ba.withdraw(9000.00), 0.001);
    		assertEquals(0.00, ba.money, 0.001);
    	}

    }
